\subsection{Sound Changes from Modern Pseudo-French}\label{subsec:sound-changes-from-modern-french}
\parheading{Legend}\par\medskip\noindent
Ṽ = nasalised vowel, Ṽ̃ = nasal vowel, V = any vowel (or, in conjunction with Ṽ/Ṽ̃, oral vowel)\\
C̃ = nasalised consonant (e.g. /ɰ̃/, but not true nasals), C = any consonant\\

\TwoCols{
\parheading{Preliminary Changes}
\begin{enum}
    \item g, r, w > ɰ ⟨r⟩
    \item œ, œ̃ > y, ỹ
    \item y > j / \_(\#)V
    \item V > $\emptyset$ / \_\#V
    \item j > ɥ ⟨y’⟩
    \item C > $\emptyset$ / \#\_C
    \item C > $\emptyset$ / C\_\#
    \item k > x\footnotemark
    \item ʃ, ʒ > ɕ ⟨ç⟩, ʑ ⟨j⟩
    \item nt > nθ
    \item t > d$_2$ (‘hard /d/’)
    \item p > b$_2$ (‘hard /b/’)
    \item f, v > ɸ, β ⟨b’h⟩
\end{enum}
}{
\parheading{Great Nasal Shift}
\begin{enum}[13]
    \item ɔl, ɔ̃l > ɰ ⟨r⟩, ɰ̃ ⟨w⟩
    \item V, Ṽ > Ṽ, Ṽ̃ / \_[C̃ɥɰ], [C̃ɥɰ]\_
    \item n, m, ŋ, ɲ, C̃ > $\emptyset$ / V\_\#
    \item ɲ, ŋ > n
    \item m, l > ʋ̃ ⟨v⟩, ɮ̃ ⟨l⟩
\end{enum}

\parheading{Intervocalic Lenition (/ V\_V is implied)}
\begin{enum}[18]
    \item x, s > h ⟨’⟩
    \item ɕ, ɮ̃ > j̊ ⟨ç⟩, ɥ̃ ⟨l⟩
    \item nθ > n
    \item d, d$_2$, b, b$_2$ > ð, θ ⟨t’h⟩, β, bʱ ⟨bh⟩
    \item e > $\emptyset$ / C\_C
\end{enum}

\parheading{Late Changes}
\begin{enum}[22]
    \item V[-nasal] > V̥ / \_\#
\end{enum}
}\medskip
