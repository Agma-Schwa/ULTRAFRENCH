\subsection{Sound Changes from Modern Pseudo-French}\label{subsec:sound-changes-from-modern-french}
\parheading{Legend}\par\medskip\noindent
Ṽ = nasalised vowel, Ṽ̃ = nasal vowel, V = any vowel (or, in conjunction with Ṽ/Ṽ̃, oral vowel)\\
C̃ = nasalised consonant (e.g. /ɰ̃/, but not true nasals), C = any consonant.\medskip

\TwoCols[.45\hsize][.45\hsize][0pt]{
\parheading{Preliminary Changes}
\begin{enum}
    \item g, r, w > ɰ ⟨r⟩
    \item œ, œ̃, ø > y, ỹ, ỹ
    \item ɔ > o
    \item y > j / \_(\#)V
    \item V > $\emptyset$ / \_\#V
    \item lj > ʎ
    \item j > ɥ ⟨y’⟩
    \item C > $\emptyset$ / \#\_C
    \item C > $\emptyset$ / C\_\#
    \item k > x\footnotemark ⟨c’h⟩
    \item ʃ, ʒ > ɕ ⟨ç⟩, ʑ ⟨j⟩
    \item nt > nθ
    \item t > d$_2$ (‘hard /d/’)
    \item p > b$_2$ (‘hard /b/’)
    \item f, v > ɸ ⟨f⟩, β ⟨b’h⟩
\end{enum}

\parheading{Great Nasal Shift}
\begin{enum}[15]
    \item Ṽl > ɰ̃ ⟨w⟩
    \item V, Ṽ > Ṽ, Ṽ̃ / \_[C̃ɥɰ], [C̃ɥɰ]\_
    \item ə̃, ə̃̃, ã, ã̃, õ, õ̃ > ɛ̃, ɛ̃̃, ɑ̃, ɑ̃̃, ɔ̃, ɔ̃̃
    \item n, m, ŋ, ɲ, C̃ > $\emptyset$ / V\_\#
    \item ɲ, ŋ > n
    \item m, l, ʎ > ʋ̃ ⟨v⟩, ɮ̃ ⟨l⟩, ʎ̝̃ ⟨l⟩
\end{enum}

\parheading{Intervocalic Lenition (/ V\_V is implied)}
\begin{enum}[21]
    \item x, s > h
    \item ɕ, ɮ̃, ʎ̝̃ > j̊ ⟨ç⟩, ɥ̃ ⟨l⟩, ɰ̃ ⟨l⟩
    \item nθ > n
    \item d, d$_2$, b, b$_2$ > ð ⟨d’h⟩, θ ⟨t’h⟩, β, bʱ ⟨bh⟩
\end{enum}

\parheading{Late Changes}
\begin{enum}[25]
    \item ə > $\emptyset$ / C\_C
    \item V[-nasal] > V̥ / \_\#
\end{enum}
